<script>
const API_URL = "https://script.google.com/macros/s/AKfycbznk6KRovt5i4lv_gMLFabkjm3j1EJJyGmdGyUB4carhyvr4eSyz3UcxTbDnoxDB_ZpKw/exec";

let current = "";
let sheetData = [];

async function loadData(){
  const res = await fetch(API_URL);
  const data = await res.json();
  sheetData = data;
  render();
}

function openPopup(booth){
  current = booth;

  let company = "";
  let name = "";

  for(let i=1;i<sheetData.length;i++){
    if(sheetData[i][0] == booth){
      company = sheetData[i][1];
      name = sheetData[i][2] || "";
      break;
    }
  }

  document.getElementById('booth').innerText = '小間：' + booth;
  document.getElementById('company').innerText = company;
  document.getElementById('name').value = name;
  document.getElementById('popup').style.display = 'block';
}

async function save(){
  const name = document.getElementById('name').value;

  await fetch(API_URL, {
    method: "POST",
    body: JSON.stringify({ booth: current, name: name })
  });

  document.getElementById('popup').style.display = 'none';
  loadData();
}

function render(){
  let html = '<tr><th>小間</th><th>出展社</th><th>担当</th></tr>';

  for(let i=1;i<sheetData.length;i++){
    let booth = sheetData[i][0];
    let company = sheetData[i][1];
    let name = sheetData[i][2] || "";

    html += `<tr><td>${booth}</td><td>${company}</td><td>${name}</td></tr>`;

    let label = document.getElementById('label-' + booth);
    if(label) label.innerText = name;
  }

  document.getElementById('list').innerHTML = html;
}

loadData();
</script>
